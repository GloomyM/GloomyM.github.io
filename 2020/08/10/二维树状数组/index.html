using namespace std;
int n,m;

const int maxn = 1030;
typedef long long ll; 
ll c[maxn][maxn];

int lowbit(int x){
    return x&(-x);
}

void update(int x,int y,ll v){
    for(int i = x ; i <= n ; i += lowbit(i)){
        for(int j = y ; j <= n ; j += lowbit(j)){
            c[i][j] += v;
        }
    }
}

ll getsum(int x,int y){
    ll ans = 0;
    for(int i = x ; i > 0 ; i -= lowbit(i)){
        for(int j = y ; j > 0 ; j -= lowbit(j)){
            ans += c[i][j];
        }
    }
    return ans;
}

}
