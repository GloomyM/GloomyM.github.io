#include<bits/stdc++.h>

using namespace std;

const int maxn = 1e4+5;
int low[maxn],instack[maxn],dfn[maxn];
int Stack[maxn],in[maxn],t[maxn],tt[maxn];
int tot,num,cnt;
int n,m;

struct node
{
    int v;
    node(int vv):v(vv){}
};
vector<node>e[maxn];

void add_edge(int u , int v)
{
    e[u].push_back(node(v));
}

void tarjan(int u)
{
    dfn[u] = low[u] = ++tot;
    Stack[++num] = u;
    instack[u] = 1;
    for(int i = 0 ; i < e[u].size() ; i++)
    {
        int v = e[u][i].v;
        if(dfn[v] == 0)
        {
            tarjan(v);
            low[u] = min(low[u],low[v]);
        }
        else if(instack[v])
        {
            low[u] = min(low[u],dfn[v]);
        }
    }
    if(dfn[u] == low[u])
    {
        while(u != Stack[num])
        {
            t[Stack[num]] = cnt;
            tt[cnt]++;
            instack[Stack[num]] = 0;
            num--;
        }
        t[Stack[num]] = cnt;
        tt[cnt]++;
        instack[Stack[num]] = 0;
        num--;
        cnt++;
    }
}


int main()
{
    cin >> n >> m;
    for(int i = 0 ; i < m ; i++)
    {
        int u,v;
        cin >> u >> v;
        add_edge(u,v);    
    }
    for(int i = 1 ; i <= n ; i++)
    {
        if(!dfn[i])
            tarjan(i);
    }
    int sum = 0;
    for(int i = 1 ; i <= n ; i++)
    {
        for(int j = 0 ; j < e[i].size() ; j++)
        {
            int v = e[i][j].v;
            if(t[i] != t[v])
                in[t[i]]++;
        }
    }
    int flag = 0;
    for(int i = 0 ; i < cnt ; i++)
    {
        if(!in[i])
        {
            flag++;
            sum = tt[i];
        }
    }
    if(flag == 1)
        cout << sum << endl;
    else
        cout << "0" << endl;
    return 0;
}
